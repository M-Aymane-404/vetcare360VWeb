  const carouselEl = document.getElementById("heroCarousel");
  const carousel = new window.bootstrap.Carousel(carouselEl, {
    interval: false,
    ride: false,
    wrap: true,
  });

  let currentIndex = 0;
  const items = carouselEl.querySelectorAll(".carousel-item");

  let resetTimeout;

  function handleHover() {
    currentIndex = (currentIndex + 1) % items.length;
    carousel.to(currentIndex);

    // Clear any existing timeout and reset
    if (resetTimeout) clearTimeout(resetTimeout);

    resetTimeout = setTimeout(() => {
      currentIndex = 0;
      carousel.to(currentIndex);
    }, 3000); // ← 3 seconds to reset
  }

  carouselEl.addEventListener("mouseenter", handleHover);

  return () => {
    carouselEl.removeEventListener("mouseenter", handleHover);
    if (resetTimeout) clearTimeout(resetTimeout);
  };
